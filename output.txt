dname=ARG1
n=ARGV[2]
l=ARGV[3]
dt=ARGV[4]

set term png
set term pngcairo size 1280,1280
set size square
set xrange [-1.5:1.5]
set yrange [-1.5:1.5]
unset key

while (n<=l){
	pngname = sprintf("../png/%s/%d.png",dname,n)
	fname = sprintf("%s/snap%05d.dat",dname,n)
	o = "u 3:4 pt 0"
	
	set output pngname
	set title sprintf("t = %d",n*dt)
	plot fname @o

	n=n+1
}
set title
